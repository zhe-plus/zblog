### 1.3 设计原则

| 原则 | 说明 |
|------|------|
| **简单优先** | 能用内置方案解决的问题不引入额外依赖 |
| **JPA 优先** | 数据库操作优先使用 Spring Data JPA |
| **DTO 隔离** | Entity 不暴露到 Controller 层 |
| **防御性编程** | 所有外部输入需校验，所有输出需转义 |
| **可测试性** | 核心业务逻辑独立于框架 |
| **渐进增强** | v1.0 表结构预留 v1.1+ 字段 |

| 术语 | 说明 |
|------|------|
| **DTO** | Data Transfer Object，数据传输对象 |
| **VO** | View Object，视图对象 |
| **BO** | Business Object，业务对象 |
| **DI** | Dependency Injection，依赖注入 |

**依赖规则**：
- Controller 只能依赖 Service 和 DTO
- Service 只能依赖 Repository、其他 Service 和工具类
- Entity 不依赖任何其他模块
- Service 之间避免循环依赖

### 2.3 应用启动流程

```
Spring Boot 启动
  → 1. 加载 application.yml 配置
  → 2. 根据 spring.profiles.active 选择数据源（sqlite/mysql/pg）
  → 3. Flyway 自动执行数据库迁移脚本
  → 4. Spring Security 初始化认证授权配置
  → 5. InstallDetectFilter 注册（检测安装状态）
  → 6. 内嵌 Tomcat 启动
  → 7. 应用就绪
```

**安装状态检测流程（InstallDetectFilter）**：

```
请求到达 → InstallDetectFilter
    │
    ├── 已安装（.zblog_installed 标记存在 或 settings 表有数据）
    │   ├── 路径 = /install → 返回 404
    │   └── 路径 ≠ /install → 正常处理
    │
    └── 未安装
        ├── 路径 = /install 或 /api/v1/install/* 或 /static/** → 正常处理
        └── 其他所有路径 → 重定向到 /install
```

### 2.4 请求处理管道

```
HTTP 请求
  → Tomcat Connector
  → InstallDetectFilter（安装状态检测）
  → Spring Security Filter Chain（认证/授权/CSRF）
  → DispatcherServlet
  → HandlerMapping（路由匹配）
  → Controller 方法
  → Service 层（业务逻辑）
  → Repository 层（数据访问）
  → 数据库 / 文件系统
```

## 3. 数据库详细设计

### 3.1 DDL 脚本（SQLite 语法 — 默认）

完整建表 DDL（13 张表 + FTS5 全文搜索索引）：

```sql
PRAGMA foreign_keys = ON;

-- 表 1：users（用户表）
CREATE TABLE IF NOT EXISTS users (
    id            INTEGER PRIMARY KEY AUTOINCREMENT,
    username      TEXT    NOT NULL UNIQUE,
    password_hash TEXT    NOT NULL,
    totp_secret   TEXT,
    created_at    TEXT    NOT NULL DEFAULT (datetime('now')),
    updated_at    TEXT    NOT NULL DEFAULT (datetime('now'))
);

-- 表 2：categories（分类表）
CREATE TABLE IF NOT EXISTS categories (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL,
    slug        TEXT    NOT NULL UNIQUE,
    description TEXT,
    sort_order  INTEGER NOT NULL DEFAULT 0,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now')),
    updated_at  TEXT    NOT NULL DEFAULT (datetime('now'))
);

-- 表 3：tags（标签表）
CREATE TABLE IF NOT EXISTS tags (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL,
    slug        TEXT    NOT NULL UNIQUE,
    description TEXT,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now')),
    updated_at  TEXT    NOT NULL DEFAULT (datetime('now'))
);

-- 表 4：posts（文章表）
CREATE TABLE IF NOT EXISTS posts (
    id            INTEGER PRIMARY KEY AUTOINCREMENT,
    title         TEXT    NOT NULL,
    slug          TEXT    NOT NULL UNIQUE,
    content       TEXT    NOT NULL,
    content_html  TEXT    NOT NULL DEFAULT '',
    summary       TEXT,
    status        TEXT    NOT NULL DEFAULT 'draft'
                  CHECK (status IN ('draft', 'private', 'public')),
    category_id   INTEGER REFERENCES categories(id) ON DELETE SET NULL,
    is_pinned     INTEGER NOT NULL DEFAULT 0,
    published_at  TEXT,
    scheduled_at  TEXT,
    created_at    TEXT    NOT NULL DEFAULT (datetime('now')),
    updated_at    TEXT    NOT NULL DEFAULT (datetime('now')),
    deleted_at    TEXT
);

CREATE INDEX idx_posts_status ON posts(status);
CREATE INDEX idx_posts_slug ON posts(slug);
CREATE INDEX idx_posts_category ON posts(category_id);
CREATE INDEX idx_posts_published ON posts(published_at);
CREATE INDEX idx_posts_deleted ON posts(deleted_at);
CREATE INDEX idx_posts_status_published ON posts(status, published_at);

-- 表 5：post_tags（文章-标签关联表）
CREATE TABLE IF NOT EXISTS post_tags (
    post_id INTEGER NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    tag_id  INTEGER NOT NULL REFERENCES tags(id) ON DELETE CASCADE,
    PRIMARY KEY (post_id, tag_id)
);
CREATE INDEX idx_post_tags_tag ON post_tags(tag_id);

-- 表 6：media（媒体文件表）
CREATE TABLE IF NOT EXISTS media (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    filename     TEXT    NOT NULL,
    storage_path TEXT    NOT NULL UNIQUE,
    mime_type    TEXT    NOT NULL,
    file_size    INTEGER NOT NULL,
    width        INTEGER,
    height       INTEGER,
    created_at   TEXT    NOT NULL DEFAULT (datetime('now')),
    deleted_at   TEXT
);

-- 表 7：post_media（文章-媒体引用表）
CREATE TABLE IF NOT EXISTS post_media (
    post_id  INTEGER NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    media_id INTEGER NOT NULL REFERENCES media(id) ON DELETE CASCADE,
    PRIMARY KEY (post_id, media_id)
);

-- 表 8：comments（评论表）
CREATE TABLE IF NOT EXISTS comments (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id      INTEGER NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    parent_id    INTEGER REFERENCES comments(id) ON DELETE CASCADE,
    author_name  TEXT    NOT NULL,
    author_email TEXT    NOT NULL,
    author_url   TEXT,
    content      TEXT    NOT NULL,
    status       TEXT    NOT NULL DEFAULT 'pending'
                 CHECK (status IN ('pending', 'approved', 'spam')),
    ip_address   TEXT,
    user_agent   TEXT,
    created_at   TEXT    NOT NULL DEFAULT (datetime('now')),
    deleted_at   TEXT
);
CREATE INDEX idx_comments_post ON comments(post_id);
CREATE INDEX idx_comments_status ON comments(status);

-- 表 9：pages（独立页面表）
CREATE TABLE IF NOT EXISTS pages (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    title        TEXT    NOT NULL,
    slug         TEXT    NOT NULL UNIQUE,
    content      TEXT    NOT NULL,
    content_html TEXT    NOT NULL DEFAULT '',
    status       TEXT    NOT NULL DEFAULT 'draft'
                 CHECK (status IN ('draft', 'public')),
    template     TEXT,
    created_at   TEXT    NOT NULL DEFAULT (datetime('now')),
    updated_at   TEXT    NOT NULL DEFAULT (datetime('now')),
    deleted_at   TEXT
);
CREATE INDEX idx_pages_slug ON pages(slug);

-- 表 10：settings（站点设置表）
CREATE TABLE IF NOT EXISTS settings (
    key        TEXT PRIMARY KEY,
    value      TEXT NOT NULL,
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);

-- 表 11：post_versions（文章版本历史表）
CREATE TABLE IF NOT EXISTS post_versions (
    id             INTEGER PRIMARY KEY AUTOINCREMENT,
    post_id        INTEGER NOT NULL REFERENCES posts(id) ON DELETE CASCADE,
    title          TEXT    NOT NULL,
    content        TEXT    NOT NULL,
    version_number INTEGER NOT NULL,
    change_summary TEXT,
    created_at     TEXT    NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_post_versions_post ON post_versions(post_id);

-- 表 12：links（友情链接表）
CREATE TABLE IF NOT EXISTS links (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    name        TEXT    NOT NULL,
    url         TEXT    NOT NULL,
    description TEXT,
    sort_order  INTEGER NOT NULL DEFAULT 0,
    is_visible  INTEGER NOT NULL DEFAULT 1,
    created_at  TEXT    NOT NULL DEFAULT (datetime('now')),
    updated_at  TEXT    NOT NULL DEFAULT (datetime('now'))
);

-- 表 13：audit_logs（操作日志表）
CREATE TABLE IF NOT EXISTS audit_logs (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    action      TEXT NOT NULL,
    target_type TEXT,
    target_id   INTEGER,
    detail      TEXT,
    ip_address  TEXT,
    created_at  TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_audit_logs_action ON audit_logs(action);
CREATE INDEX idx_audit_logs_created ON audit_logs(created_at);

-- FTS5 全文搜索虚拟表（SQLite 专用）
CREATE VIRTUAL TABLE IF NOT EXISTS posts_fts USING fts5(
    title,
    content,
    content='posts',
    content_rowid='id'
);

-- FTS 同步触发器
CREATE TRIGGER IF NOT EXISTS posts_fts_insert AFTER INSERT ON posts BEGIN
    INSERT INTO posts_fts(rowid, title, content) VALUES (new.id, new.title, new.content);
END;
CREATE TRIGGER IF NOT EXISTS posts_fts_update AFTER UPDATE ON posts BEGIN
    UPDATE posts_fts SET title = new.title, content = new.content WHERE rowid = old.id;
END;
CREATE TRIGGER IF NOT EXISTS posts_fts_delete AFTER DELETE ON posts BEGIN
    DELETE FROM posts_fts WHERE rowid = old.id;
END;
```

### 3.2 MySQL / PostgreSQL DDL 差异

| 差异项 | SQLite | MySQL | PostgreSQL |
|--------|--------|-------|------------|
| 主键自增 | `INTEGER PRIMARY KEY AUTOINCREMENT` | `BIGINT AUTO_INCREMENT PRIMARY KEY` | `BIGSERIAL PRIMARY KEY` |
| 时间默认值 | `datetime('now')` | `CURRENT_TIMESTAMP` | `NOW()` |
| 布尔值 | `INTEGER 0/1` | `TINYINT(1)` | `BOOLEAN` |
| 全文搜索 | FTS5 虚拟表+触发器 | `FULLTEXT INDEX` on posts(title, content) | `tsvector` 列 + GIN 索引 |
| TEXT | TEXT | LONGTEXT (content 字段) | TEXT |

### 3.4 JPA 实体示例（Post）

```java
@Entity
@Table(name = "posts", indexes = {
    @Index(name = "idx_posts_status", columnList = "status"),
    @Index(name = "idx_posts_slug", columnList = "slug"),
    @Index(name = "idx_posts_published", columnList = "publishedAt"),
    @Index(name = "idx_posts_status_published", columnList = "status, publishedAt")
})
public class Post {
    @Id @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false)
    private String title;

    @Column(nullable = false, unique = true)
    private String slug;

    @Column(nullable = false, columnDefinition = "TEXT")
    private String content;

    @Column(name = "content_html", nullable = false, columnDefinition = "TEXT")
    private String contentHtml;

    @Column(columnDefinition = "TEXT")
    private String summary;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false, length = 20)
    private PostStatus status = PostStatus.DRAFT;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "category_id")
    private Category category;

    @Column(name = "is_pinned", nullable = false)
    private Boolean isPinned = false;

    @Column(name = "published_at")
    private LocalDateTime publishedAt;

    @Column(name = "scheduled_at")
    private LocalDateTime scheduledAt;

    @Column(name = "created_at", nullable = false, updatable = false)
    private LocalDateTime createdAt;

    @Column(name = "updated_at", nullable = false)
    private LocalDateTime updatedAt;

    @Column(name = "deleted_at")
    private LocalDateTime deletedAt;

    @ManyToMany
    @JoinTable(name = "post_tags",
        joinColumns = @JoinColumn(name = "post_id"),
        inverseJoinColumns = @JoinColumn(name = "tag_id"))
    private Set<Tag> tags = new HashSet<>();

    @OneToMany(mappedBy = "post", cascade = CascadeType.ALL, orphanRemoval = true)
    @OrderBy("versionNumber DESC")
    private List<PostVersion> versions = new ArrayList<>();

    @PrePersist
    protected void onCreate() {
        this.createdAt = LocalDateTime.now();
        this.updatedAt = LocalDateTime.now();
        if (this.status == PostStatus.PUBLIC || this.status == PostStatus.PRIVATE) {
            this.publishedAt = LocalDateTime.now();
        }
    }

    @PreUpdate
    protected void onUpdate() {
        this.updatedAt = LocalDateTime.now();
    }
    // getters/setters ...
}
```

### 3.5 索引设计汇总

| 索引 | 表 | 字段 | 类型 | 理由 |
|------|-----|------|------|------|
| `idx_posts_status_published` | posts | status, published_at | 复合 | 前台公开文章列表（最常用查询） |
| `idx_posts_slug` | posts | slug | UNIQUE | Slug 查找 |
| `idx_posts_category` | posts | category_id | 普通 | 按分类筛选 |
| `idx_posts_published` | posts | published_at | 普通 | 按发布时间排序 |
| `idx_posts_deleted` | posts | deleted_at | 普通 | 软删除过滤 |
| `idx_post_tags_tag` | post_tags | tag_id | 普通 | 按标签查文章 |
| `idx_comments_post` | comments | post_id | 普通 | 按文章查评论 |
| `idx_comments_status` | comments | status | 普通 | 按状态筛选评论 |
| `idx_pages_slug` | pages | slug | UNIQUE | 独立页面查找 |
| `idx_post_versions_post` | post_versions | post_id | 普通 | 按文章查版本 |
| `idx_audit_logs_action` | audit_logs | action | 普通 | 按操作类型筛选 |
| `idx_audit_logs_created` | audit_logs | created_at | 普通 | 按时间排序 |
| `posts_fts` (SQLite) | posts | title, content | FTS5 | 全文搜索 |
| `FULLTEXT idx_posts_ft` (MySQL) | posts | title, content | FULLTEXT | 全文搜索 |
| `idx_posts_tsvector` (PG) | posts | tsvector列 | GIN | 全文搜索 |

### 4.1 通用响应格式

```json
{ "data": { "id": 1, "title": "...", "slug": "...", "created_at": "2026-06-16T10:30:00Z" } }
```

**成功响应（列表）**：
```json
{
  "data": [ /* ... */ ],
  "pagination": { "total": 100, "page": 1, "page_size": 10, "total_pages": 10 }
}
```

**错误响应**：
```json
{ "error": { "code": "INVALID_INPUT", "message": "标题不能为空" } }
```

## 5. 核心业务逻辑设计

### 5.1 文章状态机

**状态定义**：`DRAFT`（草稿）、`PRIVATE`（私密发布）、`PUBLIC`（公开发布）

**状态流转**：任意状态之间可互转。特殊规则：
- → `PRIVATE`/`PUBLIC` 时，若 `published_at` 为空则设为当前时间
- → `DRAFT` 时，`published_at` 保留不重置
- 任何变更：`updated_at` 自动更新
- content 变更时：`content_html` 重新生成

### 5.2 Markdown 渲染与安全清洗流程

```
Markdown 原始文本 (content)
  → 1. BacklinkService 解析 [[slug]] 双向链接
  → 2. commonmark-java 渲染 Markdown → HTML（含 GFM 扩展：表格、删除线、任务列表、脚注）
  → 3. Jsoup Safelist 白名单清洗（移除 <script>、<iframe>、onerror 等）
  → 4. 代码块增强（添加 language 类名、复制按钮容器）
  → 安全 HTML (content_html)
```

**MarkdownService 核心方法**：

```java
@Service
public class MarkdownService {
    private final Parser parser;        // CommonMark 解析器（含 GFM 扩展）
    private final HtmlRenderer renderer;
    private final BacklinkService backlinkService;

    public String renderToHtml(String markdown, Long postId) {
        // 1. 解析双向链接
        String processed = postId != null
            ? backlinkService.resolveWikiLinks(markdown, postId)
            : markdown;
        // 2. Markdown → HTML
        Node document = parser.parse(processed);
        String html = renderer.render(document);
        // 3. HTML 清洗
        return HtmlSanitizer.sanitize(html);
    }

    public String generateSummary(String markdown, int maxLength) {
        // 去除 Markdown 标记，提取纯文本，截取 maxLength 字符
        String text = markdown.replaceAll("#+\\s+", "")
            .replaceAll("\\*\\*(.+?)\\*\\*", "$1")
            .replaceAll("`{1,3}[^`]*`{1,3}", "")
            .replaceAll("!\\[.*?\\]\\(.*?\\)", "")
            .replaceAll("\\[([^\\]]+)\\]\\(.*?\\)", "$1")
            .replaceAll("\\n+", " ").trim();
        return text.length() <= maxLength ? text : text.substring(0, maxLength) + "...";
    }
}
```

### 5.3 双向链接解析算法

```
正文中的 [[other-article-slug]]
  → 1. 正则匹配 [[...]] 提取 slug 列表
  → 2. 批量查询 posts WHERE slug IN (...) AND status='public' AND deleted_at IS NULL
  → 3. 存在 → 替换为 [title](/post/slug)；不存在 → 保留原文本 [slug](/post/slug)
```

```java
@Service
public class BacklinkService {
    private static final Pattern WIKI_LINK = Pattern.compile("\\[\\[([^\\]]+)\\]\\]");

    public String resolveWikiLinks(String markdown, Long currentPostId) {
        Matcher m = WIKI_LINK.matcher(markdown);
        StringBuffer sb = new StringBuffer();
        while (m.find()) {
            String slug = m.group(1).trim();
            String replacement = postRepository
                .findBySlugAndStatusAndDeletedAtIsNull(slug, PostStatus.PUBLIC)
                .map(p -> String.format("[%s](/post/%s)", p.getTitle(), p.getSlug()))
                .orElseGet(() -> String.format("[%s](/post/%s)", slug, slug));
            m.appendReplacement(sb, Matcher.quoteReplacement(replacement));
        }
        m.appendTail(sb);
        return sb.toString();
    }
}
```

### 5.5 Slug 生成算法

```java
public class SlugUtils {
    private static final Set<String> RESERVED = Set.of(
        "search", "archive", "rss.xml", "sitemap.xml", "robots.txt",
        "feed", "admin", "api", "install", "login", "logout",
        "page", "post", "category", "tag", "uploads", "static", "health"
    );

    public static String generate(String title) {
        String slug = title.toLowerCase().trim()
            .replaceAll("[^a-z0-9\\u4e00-\\u9fff\\-\\s]", "")
            .replaceAll("[\\u4e00-\\u9fff]", "-")  // 中文→连字符
            .replaceAll("\\s+", "-")
            .replaceAll("-{2,}", "-")
            .replaceAll("^-|-$", "");
        return slug.isEmpty() ? "untitled" : slug;
    }

    public static boolean isReserved(String slug) {
        return RESERVED.contains(slug.toLowerCase());
    }
}
```

### 6.2 登录保护（频率限制）

```java
@Component
public class RateLimitFilter extends OncePerRequestFilter {
    private final ConcurrentHashMap<String, Pair<Integer, LocalDateTime>> failures = new ConcurrentHashMap<>();
    private static final int MAX_ATTEMPTS = 5;
    private static final long LOCK_MINUTES = 15;

    @Override
    protected void doFilterInternal(HttpServletRequest req, HttpServletResponse res,
                                     FilterChain chain) throws IOException, ServletException {
        if (!isLoginRequest(req)) { chain.doFilter(req, res); return; }

        String ip = getClientIp(req);
        Pair<Integer, LocalDateTime> record = failures.get(ip);

        // 检查锁定状态
        if (record != null && record.getSecond() != null
            && LocalDateTime.now().isBefore(record.getSecond())) {
            sendLockedResponse(res);
            return;
        }

        ContentCachingResponseWrapper wrapper = new ContentCachingResponseWrapper(res);
        chain.doFilter(req, wrapper);

        // 处理登录结果
        if (wrapper.getStatus() == 401) {
            failures.compute(ip, (k, v) -> {
                int count = (v == null) ? 1 : v.getFirst() + 1;
                return Pair.of(count, count >= MAX_ATTEMPTS
                    ? LocalDateTime.now().plusMinutes(LOCK_MINUTES) : null);
            });
        } else if (wrapper.getStatus() == 200) {
            failures.remove(ip); // 登录成功清除记录
        }
        wrapper.copyBodyToResponse();
    }
}
```

### 8.4 API 客户端封装

```typescript
// api/client.ts
class ApiClient {
    private csrfToken: string | null = null;

    private async request<T>(method: string, path: string, body?: unknown): Promise<T> {
        const headers: Record<string, string> = { 'Content-Type': 'application/json' };
        if (this.csrfToken) headers['X-CSRF-Token'] = this.csrfToken;

        const res = await fetch(path, { method, headers, credentials: 'same-origin',
            body: body ? JSON.stringify(body) : undefined });

        if (res.status === 401) { window.location.href = '/admin/login'; throw new Error('Unauthorized'); }

        const csrf = res.headers.get('X-CSRF-Token');
        if (csrf) this.csrfToken = csrf;

        if (!res.ok) { const err = await res.json(); throw new ApiException(err.error.code, err.error.message, res.status); }
        if (res.status === 204) return undefined as T;
        return res.json();
    }

    get<T>(path: string) { return this.request<T>('GET', path); }
    post<T>(path: string, body?: unknown) { return this.request<T>('POST', path, body); }
    put<T>(path: string, body?: unknown) { return this.request<T>('PUT', path, body); }
    patch<T>(path: string, body?: unknown) { return this.request<T>('PATCH', path, body); }
    delete<T>(path: string) { return this.request<T>('DELETE', path); }

    async upload<T>(path: string, file: File): Promise<T> {
        const fd = new FormData(); fd.append('file', file);
        const res = await fetch(path, { method: 'POST', credentials: 'same-origin',
            headers: this.csrfToken ? { 'X-CSRF-Token': this.csrfToken } : {}, body: fd });
        if (res.status === 401) { window.location.href = '/admin/login'; throw new Error('Unauthorized'); }
        if (!res.ok) { const err = await res.json(); throw new ApiException(err.error.code, err.error.message, res.status); }
        return res.json();
    }
}
export const api = new ApiClient();
```
